<script>
const messagesDiv = document.getElementById('messages');
const userInput = document.getElementById('user-input');
const sendBtn = document.getElementById('send-btn');
function addMessage(text, sender){
const msgDiv = document.createElement('div');
msgDiv.classList.add('message', sender);
msgDiv.innerHTML = text;
messagesDiv.appendChild(msgDiv);
messagesDiv.scrollTop = messagesDiv.scrollHeight;
}
function botResponse(userText){
userText = userText.toLowerCase();
let response = "Olá! Estou aqui para tirar suas dúvidas sobre o produto.";
if(userText.includes("preço") || userText.includes("quanto custa")){
response = "O preço é incrível! Quer que eu te envie o link de compra?";
} else if(userText.includes("comprar") || userText.includes("link") ||
userText.includes("quero")){
response = 'Perfeito! Clique aqui para garantir seu produto: <a
href="https://seulink.com" target="_blank">Comprar Agora</a>';
} else if(userText.includes("informação") || userText.includes("detalhes") ||
userText.includes("mais")){
response = "Claro! Esse produto é excelente e vai resolver seu problema.
Quer que eu te envie o link de compra?";
} else {
response = "Interessante! Posso te ajudar com mais informações ou te
enviar o link de compra.";
}
return response;
}
sendBtn.addEventListener('click', ()=>{
const text = userInput.value;
if(text.trim() === '') return;
addMessage(text, 'user');
userInput.value = '';
setTimeout(()=>{ addMessage(botResponse(text), 'bot'); }, 500);
});
userInput.addEventListener('keypress', function(e){
if(e.key === 'Enter') sendBtn.click();
});
